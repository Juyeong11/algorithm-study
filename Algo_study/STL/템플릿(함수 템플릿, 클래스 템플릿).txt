		pf(*begin++);

	string sep;//출력 구분자 정보

//3번
#include<iostream>

using namespace std;

template<typename T>
class Queue
{
	T buf[100];
	int top;
	int bot;
	public:
	Queue(): top(0), bot(0){}
	
	void Push(T data)
	{
		buf[top++] = data;
		
	}
	
	T Pop()
	{
		if(bot != 99) 
			return buf[bot++];
		return buf[0];
	}
	
	bool Empty()
	{
		return top == 0;	
	}
};

int main()
{
	Queue<int> st;//()의 차이가 뭐야?
	
	st.Push(10);
	st.Push(20);
	st.Push(30);
	
	if(!st.Empty())
		cout<<st.Pop()<<endl;
	if(!st.Empty())
		cout<<st.Pop()<<endl;
	if(!st.Empty())
		cout<<st.Pop()<<endl;
}
