void turn_time(int clock_time[],int switch_num, int delta)//1이면 돌리고 -1이면 반대로 돌린다
{
	for(int i=0;i<5;++i)
	{
		if(Connect_switch[switch_num][i] == -1)
			continue;
		int& t = clock_time[Connect_switch[switch_num][i]];
		
		t += 3*delta;	
		
		if(t == 15)
			t = 3;
		if(t == 0)
			t = 12;
	}
}


int set_clock(int clock_time[],int t_count[],int n)
{
	bool finsh = true;
	
	for(int i=0;i<16;++i)
		if(clock_time[i] != 12)
		{
			finsh=false;
			break;
		}
	int finsh2 = 0;
	
	for(int i=0;i<10;++i)
		finsh2 += t_count[i];
	
	if(finsh2==30)
	{
		//cout<<"f";
		return 0;
	}

	
	if(finsh)
	{
		for(int i=0;i<10;++i)
			cout<<t_count[i]<<", ";
		cout<<endl;
			
		for(int i=0;i<16;++i)
			cout<<clock_time[i]<<" ";
		cout<<endl;
		cout<<endl;
		return 1;
	}
	
	int ret = 0;
	for(int sw=n;sw<10;++sw)
	{
		//cout<<sw<<": "<<t_count[sw]<<endl;
		if(t_count[sw] >= 3)
			continue;
		
		t_count[sw]++;
		
		turn_time(clock_time,sw,1);
	
		ret += set_clock(clock_time,t_count,sw);
		
		turn_time(clock_time,sw,-1);
		
		t_count[sw]--;
	}
	return ret;
}
